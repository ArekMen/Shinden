/* === HERO: zero przycinania, skalowanie do obrazu === */
/* Proporcja Twojego banera (2048x745 ≈ 2.75:1) */
:root { --hero-ratio: 2048 / 745; }

/* Najwyższa specyficzność + !important, żeby przebić wszystko */
html body .hero-image-container{
  /* rozmiar kontenera z proporcji (działa w nowoczesnych przeglądarkach) */
  aspect-ratio: var(--hero-ratio) !important;

  /* fallback na wypadek problemów z aspect-ratio */
  height: auto !important;
  min-height: 0 !important;
  /* jeśli coś wcześniej ustawiało stałe px – skasuj to paddingiem na 36.4% szerokości */
  padding-top: 36.4% !important;
  /* i zaraz potem zerujemy, jeśli aspect-ratio zadziała – dzięki czemu nie robi się podwójna wysokość */
  padding-top: 0 !important;

  /* tło bez przycinania */
  background-size: contain !important;
  background-repeat: no-repeat !important;
  background-position: center center !important;
  box-shadow: none !important;
}

/* Jeśli czasem .hero-image-container nie istnieje – fallback baner przed treścią */

  display: block;
  width: 100% !important;
  aspect-ratio: var(--hero-ratio) !important;
  background-position: center center !important;
  background-repeat: no-repeat !important;
  background-size: contain !important;
  margin-bottom: 8px;
}
